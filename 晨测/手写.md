### 手写快排

```javascript
function quickSort(arr) {
            if (arr.length <= 1) {
                return arr
            }
            var left = [];
            var right = []
            var center = arr.pop()
            arr.forEach(function (item, index) {
                if (item <= center) {
                    left.push(item)
                } else if (item > center) {
                    right.push(item)
                }
            })
            var re = quickSort(left).concat(center, quickSort(right));
            return re
}
```
